const CarModelReducer = (
    state: CarModelState = initialState,
    action: CarModelAction
): CarModelState => {
    switch (action.type) {
        case actionTypes.ADD_CAR_MODEL:
            const newCarModel: ICarModel = {
                model: action.carModel.model,
                color: action.carModel.color,
                yearManufacture: action.carModel.yearManufacture,
                insuranceValid: action.carModel.insuranceValid,
                kms: action.carModel.kms,
                location: action.carModel.location,
                noOfOwners: action.carModel.noOfOwners,
                transmission: action.carModel.transmission,
                externalfitment: action.carModel.externalfitment,
                photo: "ford.jpeg"
            };
            return {
                ...state,
                carModels: state.carModels.concat(newCarModel)
            };

    }
    return state;
};

export default CarModelReducer;
